<body>

  <h1>Consulta de Nota Final</h1>
  <div class="container">
    <label for="grupo">Selecione o Grupo:</label>
    <select id="grupo" onchange="atualizarCampos()">
      <option value="1">Grupo 1</option>
      <option value="2">Grupo 2</option>
      <option value="3">Grupo 3</option>
      <option value="4">Grupo 4</option>
    </select>

    <form id="formNotas">
      <div id="camposNotas"></div>
      <button type="button" onclick="calcularNota()">Calcular Nota Final</button>
    </form>

    <div id="resultado"></div>
  </div>

  <script>
    // Pesos configuráveis para cada grupo
    const pesos = {
      1: { portugues: 1, ingles: 1, humanas: 1, natureza: 1, matematica: 1, redacao: 1 },
      2: { portugues: 1, ingles: 1, humanas: 1, natureza: 1, matematica: 1, redacao: 1 },
      3: { portugues: 1, ingles: 1, humanas: 1, natureza: 1, matematica: 1, redacao: 1 },
      4: { portugues: 1, ingles: 1, humanas: 1, natureza: 1, biologia: 1, matematica: 1, redacao: 1 }
    };
    // ↑ Edite aqui os pesos de cada matéria ↑

    function atualizarCampos() {
      const grupo = document.getElementById("grupo").value;
      const camposDiv = document.getElementById("camposNotas");
      camposDiv.innerHTML = "";

      const materias = grupo == 4 
        ? ["Português", "Inglês", "Ciências Humanas", "Ciências da Natureza", "Biologia", "Matemática", "Redação"]
        : ["Português", "Inglês", "Ciências Humanas", "Ciências da Natureza", "Matemática", "Redação"];

      materias.forEach(mat => {
        const id = mat.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/\s+/g, "");
        camposDiv.innerHTML += `
          <label for="${id}">${mat} (0 a 10):</label>
          <input type="number" id="${id}" min="0" max="10" step="0.1" placeholder="Digite sua nota">
        `;
      });
    }

    function calcularNota() {
      const grupo = document.getElementById("grupo").value;
      const materias = Object.keys(pesos[grupo]);
      let somaPonderada = 0;
      let somaPesos = 0;

      materias.forEach(mat => {
        const valor = parseFloat(document.getElementById(mat).value) || 0;
        somaPonderada += valor * pesos[grupo][mat];
        somaPesos += pesos[grupo][mat];
      });

      // Escala final de 0 a 100
      const notaFinal = somaPesos > 0 ? (somaPonderada / somaPesos) * 10 : 0;
      document.getElementById("resultado").innerText = `Nota Final: ${notaFinal.toFixed(2)} / 100`;
    }

    atualizarCampos(); // inicia com grupo 1
  </script>
